# 86. 分隔链表

> [题目链接](https://leetcode.cn/problems/partition-list/?envType=study-plan-v2&envId=top-interview-150)


## 维护两个链表、最后合并即可

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        ListNode *la = new ListNode();
        ListNode *la_head = la;
        ListNode *sh = new ListNode();
        ListNode *ans_head = sh;
        while (head != nullptr) {
            if (head->val < x) {
                sh->next = head;
                sh = sh->next;
            } else {
                la->next = head;
                la = la->next;
            }
            head = head->next;
        }
        // la->next为nullptr, 链表末尾
        la->next = nullptr;
        // sh->next链接la
        sh->next = la_head->next;
        return ans_head->next;
    }
};
```
