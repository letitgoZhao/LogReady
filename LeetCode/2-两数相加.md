# 2. 两数相加

> [题目链接](https://leetcode.cn/problems/add-two-numbers/?envType=study-plan-v2&envId=top-interview-150)


## 初次尝试——迭代遍历进位

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* ans = new ListNode();
        ListNode* copy = ans;
        int sum = 0, val_update = 0, carry = 0;
        while (l1 != nullptr || l2 != nullptr) {
            // 处理sum数据
            if (l1 == nullptr)
                sum = l2->val + carry;
            else if (l2 == nullptr)
                sum = l1->val + carry;
            else
                sum = l1->val + l2->val + carry;
            // 处理变更数据
            val_update = sum % 10;
            carry = (sum >= 10) ? 1 : 0;
            // 挂载新结点
            ans->next = new ListNode(val_update);
            ans = ans->next;
            // 更新指针
            if (l1 != nullptr)
                l1 = l1->next;
            if (l2 != nullptr)
                l2 = l2->next;
        }
        // 是否有进位
        if (carry) {
            ans->next = new ListNode(1);
        }

        return copy->next;
    }
};
```
