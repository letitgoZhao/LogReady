# 66. 加一

> [题目链接](https://leetcode.cn/problems/plus-one/?envType=study-plan-v2&envId=top-interview-150)

## 初次尝试——逆序遍历

综合考虑各种情况即可，不要遗漏999这种特殊情况。

```cpp
class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        // digits不为空
        int size = digits.size();
        int end = digits[size - 1];
        vector<int> ans1 = digits;
        vector<int> ans2(size + 1, 0);
        if (end < 9) {
            ans1[size - 1]++;
        } else {
            // 处理进位
            // 情况1. 239 -> 240,  [2, 4, 0]
            // 情况2. 999 -> 1000, [1, 0, 0, 0]
            int flag = 1;
            ans1[size - 1] = 0;
            for (int i = size - 2; i >= 0; i--) {
                int temp = ans1[i] + flag;
                ans1[i] = temp % 10;
                if (temp == 10)
                    flag = 1;
                else
                    flag = 0;
            }
            // 处理情况2
            if (flag) {
                ans2[0] = 1;
                ans1 = ans2;
            }
        }
        return ans1;
    }
};
```
