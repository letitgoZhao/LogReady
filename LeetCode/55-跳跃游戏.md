# 55. 跳跃游戏

> [题目链接](https://leetcode.cn/problems/jump-game/?envType=study-plan-v2&envId=top-interview-150)

## 贪心算法

```cpp
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n = nums.size();
        int right_most = 0;
        // 实时维护right_most, 直至right_most >= nums.size() - 1
        for (int i = 0; i < n; ++i) {
            if (i <= right_most) {
                // i + nums[i]表示最远可以到达的索引
                right_most = max(right_most, i + nums[i]);
                if (right_most >= n - 1)
                    return true;
            }
        }
        return false;
    }
};
```

## 动态规划

```cpp
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int sz = nums.size();
        // dp[i]表示能否到达i位置
        vector<bool> dp(sz, false);
        dp[0] = true;

        for (int i = 1; i < sz; i++) {
            for (int j = i - 1; j >= 0; j--) {
                // 是否可以到达下一位置i?
                // 状态转移方程如下:
                // dp[i] = true, if dp[j] && (i - j) <= nums[j], 其中j为[0, i-1]
                // dp[i] = false, if j不为[0, i - 1], 即默认false
                if (dp[j] == true && (i - j) <= nums[j]) {
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[sz - 1];
    }
};
```
