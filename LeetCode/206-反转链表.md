# 206. 反转链表

> [题目链接](https://leetcode.cn/problems/reverse-linked-list/)

## 头插法

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        // 方法1 - 头插法
        if (head == nullptr)
            return nullptr;
        ListNode* dummy = new ListNode(-1);

        while (head != nullptr) {
            ListNode* nex = head->next;
            head->next = dummy->next;
            dummy->next = head;
            head = nex;
        }
        ListNode* ans = dummy->next;
        delete dummy;
        return ans;
    }
};
```

## 迭代方法

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        // 方法2. 迭代方法
        // 初始：1 -> 2 -> nullptr
        // [1] nullptr <- 1
        // [2] nullptr <- 1 <- 2
        // 结束
        if (head == nullptr)
            return nullptr;
        ListNode* prev = nullptr;
        ListNode* cur = head;
        while (cur != nullptr) {
            ListNode* nex = cur->next;
            cur->next = prev;
            prev = cur;
            cur = nex;
        }
        return prev;
    }
};
```

## 递归方法

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        // 递归方法
        if (head == nullptr || head->next == nullptr) {
            return head;
        }
        // 以 1 -> 2 -> 3 -> 4 来举例
        // [1] 一直递推找到 '4' 结点, 作为新的头节点.
        // [2] 回溯至head = '3', nullptr <- 3 <- 4, newHead不变.
        // [3] 回溯至head = '2', nullptr <- 2 <- 3 <- 4, newHead不变.
        // [4] 回溯至head = '1', nullptr <- 1 <- 2 <- 3 <- 4, newHead不变.
        // 调用最初位置, 回溯完成, 返回newHead即可完成反转操作.

        // 递归找到新的头结点
        ListNode* newHead = reverseList(head->next);
        head->next->next = head;
        head->next = nullptr;
        return newHead;
    }
};
```
