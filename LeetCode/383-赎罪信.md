# 383.赎金信

> [LeetCode题目链接](https://leetcode.cn/problems/ransom-note/?envType=study-plan-v2&envId=top-interview-150)

## vector容器实现字符统计

```cpp
class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        // magazine中的字符来构建ransomNote, 且magazine的每个字符仅能使用一次
        // magazine的字母('a'-'z')的各个统计次数均大于ransomNote中相同字母的统计次数
        if (ransomNote.size() > magazine.size())
            return false;
        vector<int> cnt(26);
        for (const auto& ch : magazine) {
            cnt[ch - 'a']++; // 统计'a'-'z'次数
        }

        for (const auto& ch : ransomNote) {
            cnt[ch - 'a']--; // 使用magazine构造
            if (cnt[ch - 'a'] < 0)
                return false; // magazine中对应的'a'-'z'不足
        }
        return true; // 可以用magazine构造
    }
};
```

## unordered_map实现字符统计

```cpp
class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        if (ransomNote.size() > magazine.size())
            return false;
        std::unordered_map<char, int> cnt;

        for (const auto& ch : magazine) {
            cnt[ch]++;
        }
        for (const auto& ch : ransomNote) {
            cnt[ch]--;
            if (cnt[ch] < 0)
                return false;
        }
        return true;
    }
};
```
