# 45. 跳跃游戏II

> [题目链接](https://leetcode.cn/problems/jump-game-ii/?envType=study-plan-v2&envId=top-interview-150)

## 动态规划

```cpp
class Solution {
public:
    int jump(vector<int>& nums) {
        int sz = nums.size();
        // 如果仅有一个元素, 不用跳, 因而返回0
        if (sz == 1)
            return 0;
        vector<int> dp(sz, 0);
        // ** dp[i]表示从起始位置0 跳到终止位置i 的最小跳跃次数 **
        dp[0] = 1;
        // 状态转移方程为:
        // 1. dp[i] = dp[j] + 1, if nums[j] + j >= i
        // 表示存在一个j, 使得起始 -> 过渡结点j -> 结束
        // 2. dp[i] = 1, if nums[0] >= i, 表示起始跳到结束
        for (int i = 1; i < sz; ++i) {
            // 情况1处理
            if (nums[0] >= i) {
                dp[i] = 1;
                continue;
            }
            // 情况2处理
            for (int j = 0; j <= i - 1; j++) {
                if (nums[j] + j >= i) {
                    dp[i] = dp[j] + 1;
                    break;
                }
            }
        }
        return dp[sz - 1];
    }
};
```
