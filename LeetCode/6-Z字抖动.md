# 6. Z字抖动*

> [题目链接](https://leetcode.cn/problems/zigzag-conversion/description/?envType=study-plan-v2&envId=top-interview-150) 


## 模拟
```cpp
class Solution {
public:
    string convert(string s, int numRows) {
        if (numRows < 2)
            return s;
        vector<string> rows(numRows, "");
        int i = 0, flag = -1;
        // 分别得到每行的字符串
        for (auto c : s) {
            // *核心代码：开头或结尾需要反转方向来填充rows[i]*
            if (i == 0 || i == numRows - 1)
                flag = -flag;
            rows[i].push_back(c);
            // i = i + flag确定添加到哪个rows中
            i = i + flag;
        }
        string ans_builder = "";
        for (const string& row : rows) {
            ans_builder += row;
        }
        return ans_builder;
    }
};
```
