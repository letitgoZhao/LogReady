# 238. 除自身以外数组的乘积

> [题目链接](https://leetcode.cn/problems/product-of-array-except-self/?envType=study-plan-v2&envId=top-interview-150)

## 划分范围、左右求值

```cpp
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int sz = nums.size();
        // L、R分别表示左右两侧的乘积列表
        vector<int> L(sz, 0), R(sz, 0);
        vector<int> answer(sz);
        // 处理L数组, L[i]表示i前所有数字的乘积, 不包含i
        L[0] = 1;
        for (int i = 1; i < sz; ++i) {
            L[i] = L[i - 1] * nums[i - 1];
        }
        // 处理R数组, R[i]表示i后所有数字的乘积, 不包含i
        R[sz - 1] = 1;
        for (int i = sz - 2; i >= 0; --i) {
            R[i] = R[i + 1] * nums[i + 1];
        }
        // 处理answer数组
        for (int i = 0; i < sz; ++i) {
            answer[i] = L[i] * R[i];
        }
        return answer;
    }
};
```
