# 9. 回文数

> [题目链接](https://leetcode.cn/problems/palindrome-number/?envType=study-plan-v2&envId=top-interview-150)

## 初次尝试

```cpp
class Solution {
public:
    bool isPalindrome(int x) {
        // 处理负数情况
        if (x < 0)
            return false;
        // 处理非负情况
        vector<char> nums;
        // 121 -> '1'-'2'-'1'
        // 234 -> '4'-'3'-'2'
        while (x != 0) {
            int elem = x % 10;
            nums.emplace_back(elem + '0');
            x /= 10;
        }
        int size = nums.size();
        int left = 0, right = size - 1;
        while (left <= right) {
            if (nums[left] != nums[right]) 
                return false;
            left++, right--;
        }
        // left == right
        return true;
    }
};
```

## 官方题解——优化算法

```cpp
class Solution {
public:
    bool isPalindrome(int x) {
        // 1. 当x为负数时, 不是回文数.
        // 2. 当x的个位数字为0时, 不是回文数.
        if (x < 0 || (x != 0 && x % 10 == 0))
            return false;
        // 3. 处理一般情况, 判断是否是回文数
        int reverseNum = 0;
        // 121 -> 回文数
        // reverseNum = 0 * 10 + 1  = 1, x = 12
        // reverseNum = 1 * 10 + 2  = 12, x = 1
        // reverseNum / 10 == x 情况.
        while (x > reverseNum) {
            reverseNum = reverseNum * 10 + x % 10;
            x /= 10;
        }
        // 1221 -> 回文数
        // reverseNum = 0 * 10 + 1 = 1, x = 122
        // reverseNum = 1 * 10 + 2 = 12, x = 12
        // x == reverseNum 情况.
        return (x == reverseNum) || (reverseNum / 10 == x);
    }
};
```
