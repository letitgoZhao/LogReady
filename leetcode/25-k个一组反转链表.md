# 25. K个一组翻转链表

> [题目链接](https://leetcode.cn/problems/reverse-nodes-in-k-group/?envType=study-plan-v2&envId=top-interview-150)

## 模拟+翻转链表原型题
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode* hair = new ListNode(-1, head);

        // prev作为每组链表的起始边界条件, nex作为每组链表的结束边界条件.
        // head作为每组链表的头结点, tail作为每组链表的尾结点.
        // [-1] -> 1 -> 2 -> 3 -> 4 -> 5, k = 2

        ListNode* prev = hair;

        while (head != nullptr) {
            /* 1. 找到nex边界条件 */
            ListNode* tail = prev;
            // k个一组进行翻转
            for (int i = 0; i < k; ++i) {
                tail = tail->next;
                if (tail == nullptr) {
                    // **说明k很大, 而链表长度较小, 不反转链表**
                    return hair->next;
                }
            }
            // tail一定不为nullptr
            ListNode* nex = tail->next;

            /* 2. 翻转head-tail链表 */
            pair<ListNode*, ListNode*> result = reverseLists(head, tail);
            head = result.first;
            tail = result.second;

            /* 3. 链接边界条件 prev 和 nex */
            prev->next = head;
            tail->next = nex;

            /* 4. 更新边界条件、迭代条件 */
            prev = tail;
            head = tail->next;
        }
        return hair->next;
    }

private:
    pair<ListNode*, ListNode*> reverseLists(ListNode* head, ListNode* tail) {
        ListNode* prev = nullptr;
        // 保证head不被污染
        ListNode* p = head;
        // 1. 典型的反转链表题目
        // 1 -> 2 -> 3 -> nullptr
        // nullptr <- 1 <- 2 <- 3

        // 2. 此题目所用的反转链表
        // 1 -> [2 -> 3] -> 4
        // 1 -> [2 <- 3] -> 4

        // 因而copy反转链表实现思路即可

        while (prev != tail) {
            ListNode* nex = p->next;
            p->next = prev;
            prev = p;
            p = nex;
        }
        return {tail, head};
    }
};
```
