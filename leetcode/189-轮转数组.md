# 189. 轮转数组

> [题目链接](https://leetcode.cn/problems/rotate-array/?envType=study-plan-v2&envId=top-interview-150)

## 初次尝试——维护两个vector

```cpp
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        if (k == 0) return;
        // [1, 2, 3, 4, 5, 6, 7] -> vc2[5, 6, 7] - vc1[1, 2, 3, 4]
        int sz = nums.size();
        int valid = k % sz;
        // 维护两个vector即可
        vector<int> vc1, vc2;
        for (int i = 0; i < sz; ++i) {
            // 修改后, vc1为后边, vc2为前边
            (i < sz - valid) ? vc1.push_back(nums[i]) : vc2.push_back(nums[i]);
        }

        for (int i = 0; i < sz; ++i) {
            (i < valid) ? nums[i] = vc2[i] : nums[i] = vc1[i - valid];
        }
    }
};
```

## 数组翻转

```cpp
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        k = k % nums.size();

        reverse(nums, 0, nums.size() - 1);
        reverse(nums, 0, k - 1);
        reverse(nums, k, nums.size() - 1);
    }
private:
    void reverse(vector<int>& nums, int start, int end) {
        while (start < end) {
            swap(nums[start], nums[end]);
            start++, end--;
        }
    }
};
```

                nums[size - 1] = lose;
