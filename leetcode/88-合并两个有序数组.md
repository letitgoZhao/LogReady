# 88. 合并两个有序数组

> [题目链接](https://leetcode.cn/problems/merge-sorted-array/description/?envType=study-plan-v2&envId=top-interview-150)

## 快速排序

```cpp
class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        for (int i = 0; i != n; ++i) {
            nums1[m + i] = nums2[i];
        }
        sort(nums1.begin(), nums1.end());
    }
};
```

## 双下表检索-倒序检索

```cpp
class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        int p1 = m - 1;
        int p2 = n - 1;
        int tail = m + n - 1, temp;

        // nums1=[], nums2=[1,2,3]时
        // nums1=[1,2,3], nums2=[]时
        // nums1=[1,2,3], nums2=[2,3,5]时

        while (p1 >= 0 || p2 >= 0) {
            if (p1 == -1) {
                temp = nums2[p2--]; // 处理nums1数组为[]
            } else if (p2 == -1) {
                temp = nums1[p1--]; // 处理nums2数组为[]
            } else {
                // 将nums1数组和nums数组中较大的值送至tail处
                temp = (nums1[p1] > nums2[p2]) ? nums1[p1--] : nums2[p2--];
            }

            nums1[tail--] = temp;
        }
    }
};
```
